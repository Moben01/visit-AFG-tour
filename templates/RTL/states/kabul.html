{% trans "درباره گردشگری کابل" %}

{% trans "کابل شهری با تاریخ غنی و فرهنگ زنده است، با بازارهای قدیمی، کوه‌های باشکوه، موسیقی سنتی، غذاهای خوش‌طعم، قالی‌های رنگارنگ و مردمی مهربان. این شهر تجربه‌ای شگفت‌انگیز برای مسافرانی که به دنبال تاریخ، فرهنگ، ماجراجویی و مهمان‌نوازی اصیل افغان هستند ارائه می‌دهد. واقع در دره رودخانه کابل، این شهر برای علاقه‌مندان به تاریخ، کاوشگران فرهنگی، دوستداران طبیعت و تعطیلات خانوادگی عالی است." %}

{% trans "شکوه معماری: قلعه‌ها، مساجد و خانه‌های سنتی!" %}

{% trans "جاذبه‌های گردشگری اصلی کابل شامل قلعه تاریخی بالا حصار، کاخ‌های سلطنتی مانند دارالامان و خانه‌های سنتی افغان است که میراث غنی شهر را نشان می‌دهند. قلعه بالا حصار که بر تپه‌ای مشرف به کابل واقع شده است، نگاهی به قرن‌ها تاریخ افغانستان ارائه می‌دهد. اگرچه کابل کاخ‌های بزرگ مانند برخی مناطق ندارد، معماری اسلامی، بازارهای پرجنب و جوش و باغ‌های باستانی تجربه‌ای فرهنگی منحصر به فرد فراهم می‌کنند. بازدیدکنندگان می‌توانند مساجد و زیارتگاه‌های باشکوهی را نیز ببینند که اهمیت تاریخی و معنوی شهر را نشان می‌دهند." %}

{% trans "ترکیبی غنی از رقص‌های سنتی، آهنگ‌های دلنشین، سواری با اسب و رنگ‌های زنده!" %}

{% trans "برای علاقه‌مندان به فرهنگ، کابل جشنواره‌ها و نمایشگاه‌های زنده‌ای برای تجربه سنت‌های افغان ارائه می‌دهد. جشن‌های شاد نوروز با موسیقی، رقص و گردهمایی‌های رنگارنگ فرا رسیدن بهار را نشان می‌دهند. عروسی‌های سنتی و جشن‌های مذهبی مانند عید، آداب و رسوم غنی، رقص‌های محلی مانند آتان و موسیقی دلنشین را به نمایش می‌گذارند. بازدیدکنندگان می‌توانند بازارها و نمایشگاه‌های اسب را نیز تجربه کنند که میراث عمیق کوچ‌نشینی و روستایی افغانستان را منعکس می‌کنند." %}

{% trans "کوه‌های باشکوه افغانستان و حیات وحش منحصر به فرد را کاوش کنید." %}

{% trans "افغانستان با کوه‌های باشکوه و مناظر ناهموار، فرصت‌های شگفت‌انگیزی برای گردشگری حیات وحش ارائه می‌دهد. این کشور میزبان گونه‌های کمیاب و در خطر انقراض مانند پلنگ برفی، قوچ مارکوپولو، جبیر و خرس‌ها است. پارک‌های ملی مانند بند امیر — اولین پارک ملی افغانستان — زیبایی طبیعی خیره‌کننده و زیستگاه گونه‌های گیاهی و جانوری متنوع را فراهم می‌کنند. برای علاقه‌مندان به پرنده‌نگری، افغانستان گونه‌های مهاجر و بومی متعددی از جمله لک‌لک‌ها و عقاب‌ها در دریاچه‌ها و تالاب‌ها دارد. اگرچه کابل شهری شلوغ است، اما به عنوان دروازه‌ای به این مناطق طبیعی بکر عمل می‌کند که طبیعت‌دوستان می‌توانند حیات وحش و زیبایی طبیعی کشور را مشاهده کنند." %}

{% trans "خرید: از جواهرات تا صنایع دستی سنتی!" %}

{% trans "خرید صنایع دستی اصیل بخش مهمی از تجربه فرهنگ افغان است. کابل به قالی‌ها و فرش‌های دست‌ساز نفیس، منسوجات رنگارنگ گلدوزی شده و جواهرات زیبا با سنگ‌های قیمتی مانند لاجورد و فیروزه مشهور است. نقاشی‌های مینیاتوری افغانستان که تحت تأثیر سبک‌های فارسی و مغولی هستند، هنر پیچیده‌ای را نشان می‌دهند و از مجموعه‌های ارزشمند محسوب می‌شوند. بازدیدکنندگان همچنین می‌توانند سفال‌های زیبا، محصولات چرمی سنتی و ابزار موسیقی مانند رباب را پیدا کنند که میراث فرهنگی غنی کشور را منعکس می‌کنند. بازارهایی مانند خیابان معروف چیکن کابل تجربه خرید زنده‌ای ارائه می‌دهند که مسافران می‌توانند سوغاتی‌های منحصر به فرد و صنایع دستی افغان اصیل را به خانه ببرند." %}

{% trans "از کبابی پلاؤ تا منتو: از غذای غنی کابل لذت ببرید!" %}

{% trans "با تأثیر تبادلات فرهنگی قرن‌ها و جغرافیای متنوع افغانستان، غذاهای افغان تجربه‌ای منحصر به فرد و خوش‌طعم ارائه می‌دهند. از غذای ملی کبابی پلاؤ — برنج معطر با کشمش، هویج و گوشت گوسفند — تا خوراکی‌های خیابانی خوشمزه مانند منتو، غذاهای افغان هر ذائقه‌ای را راضی می‌کنند. نان‌های مختلف مانند نان و بولانی همراه با خورش‌ها و کباب‌های لذیذ از گوشت گوسفند، گاو و مرغ سرو می‌شوند. همراه با ماست، سبزیجات تازه و چاشنی‌های خوش‌طعم، غذا کامل می‌شود. برای علاقه‌مندان به شیرینی، دسرهای سنتی مانند باقلوا، فرنی و شیر خرما پایان لذت‌بخشی برای هر وعده غذایی ارائه می‌دهند." %}

{% trans "تعطیلات لوکس و مهمان‌نوازی بی‌زمان افغان!" %}

{% trans "برای تعطیلات لوکس، کابل هتل‌ها و مهمان‌خانه‌های میراثی را ارائه می‌دهد که راحتی مدرن را با مهمان‌نوازی اصیل افغان ترکیب می‌کنند. اگرچه قطارهای سلطنتی وجود ندارند، مسافران می‌توانند در هتل‌هایی با دکوراسیون سنتی افغان، مهمان‌نوازی گرم و غذاهای محلی لذیذ اقامت داشته باشند. بسیاری از ساختمان‌های میراثی و کاخ‌ها بازسازی شده‌اند تا اقامتگاه‌های منحصر به فردی ارائه دهند که تاریخ و فرهنگ غنی افغانستان را نشان می‌دهند. از فضاهای داخلی زیبا با کنده‌کاری‌های چوبی پیچیده و فرش‌ها گرفته تا باغ‌های مرتب و حیاط‌های آرام، این مکان‌ها تجربه‌ای همه‌جانبه از سلطنت و سنت افغان ارائه می‌دهند. اقامت لوکس در کابل راحتی، غنای فرهنگی و استقبال گرم ارائه می‌دهد که هر مهمان را مانند سلطنت احساس می‌کند." %}

{% trans "از کوهنوردی تا پاراگلایدینگ، ماجراجویی در افغانستان انتظار شما را می‌کشد!" %}
